Ski Areas

geoiq_ski_areas = new gvector.GeoIQ({
    dataset: 164880,
    uniqueField: "NAME",
    scaleRange: [6, 20],
    infoWindowTemplate: '<div class="iw-content"><h3>{NAME}</h3></div>',
    singleInfoWindow: true,
    symbology: {
        type: "single",
        vectorOptions: {
    	    icon: "../../docs-demo/img/markers/ski-lift.png"
        }
    }
});
